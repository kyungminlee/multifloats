cmake_minimum_required(VERSION 3.27)

# Pin the compilers to Homebrew GCC 15 by default. Apple Clang lacks
# __float128 / libquadmath, and Apple-shipped LLVM Flang on Apple Silicon
# does not enable REAL(KIND=16). Override on the command line with
# -DCMAKE_CXX_COMPILER=... / -DCMAKE_Fortran_COMPILER=...
if(NOT DEFINED CMAKE_CXX_COMPILER)
    find_program(_MF_GXX NAMES g++-15 g++-14 g++-13
        PATHS /opt/homebrew/bin /usr/local/bin /usr/bin)
    if(_MF_GXX)
        set(CMAKE_CXX_COMPILER ${_MF_GXX})
    endif()
endif()
if(NOT DEFINED CMAKE_Fortran_COMPILER)
    find_program(_MF_GFORTRAN NAMES gfortran-15 gfortran-14 gfortran-13
        PATHS /opt/homebrew/bin /usr/local/bin /usr/bin)
    if(_MF_GFORTRAN)
        set(CMAKE_Fortran_COMPILER ${_MF_GFORTRAN})
    endif()
endif()

# Guard against stale src/dd_constants.hh. The header is generated by
# scripts/gen_constants.py (mpmath) and committed. Running --check diffs
# the current header against what the script would produce right now —
# if scripts/gen_constants.py was edited without rerunning it, this
# surfaces that drift as a ctest failure. Prefer `uv run --with mpmath`
# so the check works even on systems where mpmath is not installed
# globally; fall back to plain python3 if uv is unavailable (user must
# then have mpmath installed, or the test will fail).
find_program(UV_EXECUTABLE uv)
find_package(Python3 COMPONENTS Interpreter QUIET)
if(UV_EXECUTABLE)
    add_test(NAME dd_constants_up_to_date
        COMMAND ${UV_EXECUTABLE} run --with mpmath python3
                scripts/gen_constants.py --check
        WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
elseif(Python3_Interpreter_FOUND)
    add_test(NAME dd_constants_up_to_date
        COMMAND ${Python3_EXECUTABLE}
                ${CMAKE_CURRENT_SOURCE_DIR}/scripts/gen_constants.py --check
        WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
endif()

# Fortran <-> C ABI sync: every `bind(c, name=*dd*)` declaration in the
# generated Fortran module must match a MULTIFLOATS_API entry in
# multifloats.h. See scripts/check_fortran_abi_sync.sh for the rule and
# the (intentional) asymmetry — the C header is a superset because
# Fortran reimplements basic arithmetic natively.
add_test(NAME fortran_abi_sync
    COMMAND bash ${CMAKE_CURRENT_SOURCE_DIR}/scripts/check_fortran_abi_sync.sh
            ${CMAKE_CURRENT_SOURCE_DIR}/include/multifloats.h
            ${CMAKE_CURRENT_BINARY_DIR}/fsrc/generated/multifloats.f90)

# C++ tests — use __float128 / libquadmath as reference (GCC only).
# Test names are prefixed precision_* / fuzz_* / perf_* so that
# `ctest -R precision_` or `ctest -R fuzz_` picks one category at a
# time. Executable target names are unchanged (no rename cost for
# downstream tooling / CI configs).

# Helpers to cut boilerplate for the ~10 test/bench executables below.
# The compile flags and library set are identical per-language, so a
# helper is strictly easier to maintain than a dozen copies.
function(mf_add_cpp_executable name source)
    add_executable(${name} ${source})
    target_compile_options(${name} PRIVATE -O3 -fext-numeric-literals)
    # Tests are privileged consumers — they exercise src/ implementation
    # helpers (multifloats_td.hh) that aren't part of the installed API.
    target_include_directories(${name} PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src)
    target_link_libraries(${name} PRIVATE multifloats quadmath)
endfunction()

function(mf_add_fortran_executable name source)
    cmake_parse_arguments(PARSE_ARGV 2 ARG "" "" "LINK")
    add_executable(${name} ${source})
    target_compile_options(${name} PRIVATE -O3)
    target_link_libraries(${name} PRIVATE multifloatsf ${ARG_LINK})
endfunction()

# Seed-determinism: the fuzz RNG must be byte-identical across runs
# given the same (iterations, seed). Catches any non-deterministic
# state (e.g. a rogue static, uninitialized scratch, or reliance on
# system entropy). Uses small iteration count to keep ctest fast.
add_test(NAME fuzz_cpp_determinism
    COMMAND bash -c
    "diff <($<TARGET_FILE:cpp_fuzz> 2000 0xc0ffee) <($<TARGET_FILE:cpp_fuzz> 2000 0xc0ffee)")

# fortran_fuzz is generated from a fypp template. The .fypp file uses
# CHK1 / CHK2 / CHK_OP / CHKC1 / CHKC_OP macros to collapse per-op

    OUTPUT ${_FUZZ_F90}
    COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_CURRENT_BINARY_DIR}/test
    COMMAND ${FYPP_EXECUTABLE}
        ${CMAKE_CURRENT_SOURCE_DIR}/test/fuzz.fypp
        ${_FUZZ_F90}
    DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/test/fuzz.fypp
    VERBATIM)

add_test(NAME fuzz_fortran COMMAND fortran_fuzz)

    "diff <($<TARGET_FILE:fortran_fuzz>) <($<TARGET_FILE:fortran_fuzz>)")

# ABI equivalence: the two DD entry points (native operator / C wrapper)
# must produce bit-identical HI and near-identical LO (4 dp ULPs) on
# the same inputs. Catches ABI drift between the Fortran-native operator
# path and the C-ABI kernels in multifloats_math.cc.
mf_add_fortran_executable(fortran_abi_equivalence ./test/abi_equivalence.f90)
add_test(NAME precision_abi_equivalence COMMAND fortran_abi_equivalence)

# Cross-language differential fuzz. Drives the C++ and Fortran kernels
# of add/sub/mul/div/neg/abs/sqrt/{==,!=,<} from a single C++ binary on
# matched random inputs and requires bit-for-bit agreement on both
# limbs. Exists because MPFR / libquadmath oracles only certify each
# side within DD precision independently — they cannot catch sub-ULP
# algorithmic drift between the two production code paths. Transcendentals
# delegate to C++ on the Fortran side (see C_DELEGATE_UNARY_MAP in
# fsrc/multifloats.fypp) and are covered by test/abi_equivalence.f90.
add_library(crosscheck-bindings OBJECT ./test/crosscheck_bindings.f90)
target_compile_options(crosscheck-bindings PRIVATE -O3)
target_link_libraries(crosscheck-bindings PRIVATE multifloatsf)
set_target_properties(crosscheck-bindings PROPERTIES
    Fortran_MODULE_DIRECTORY ${CMAKE_BINARY_DIR}/fmod/test)

add_executable(cpp_crosscheck ./test/crosscheck.cc)
target_compile_options(cpp_crosscheck PRIVATE -O3 -fext-numeric-literals)
target_include_directories(cpp_crosscheck PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src)
target_link_libraries(cpp_crosscheck PRIVATE
    multifloats crosscheck-bindings multifloatsf quadmath)
# Mixed C++/Fortran link. Main is .cc (so g++ is the default linker
# driver) but the executable embeds Fortran object code via
# crosscheck-bindings / multifloatsf, which pulls in the gfortran
# runtime. Switching to the Fortran linker driver handles -lgfortran /
# -lquadmath automatically; libstdc++ is added explicitly since the
# Fortran driver doesn't link it by default.
set_target_properties(cpp_crosscheck PROPERTIES LINKER_LANGUAGE Fortran)
target_link_libraries(cpp_crosscheck PRIVATE stdc++)
add_test(NAME crosscheck_cpp_fortran COMMAND cpp_crosscheck 5000 0xc0ffee)

# =============================================================================
# Optional MPFR-based precision tests
# =============================================================================
if(BUILD_MPFR_TESTS)
    find_package(MPFR REQUIRED)

    # mpreal — single-header C++ wrapper around MPFR. Try system install
    # first; fall back to fetching from GitHub if the header is missing.
    find_path(MPREAL_INCLUDE_DIR mpreal.h)
    if(NOT MPREAL_INCLUDE_DIR)
        include(FetchContent)
        FetchContent_Declare(mpreal
            GIT_REPOSITORY https://github.com/advanpix/mpreal.git
            GIT_TAG        mpfrc++-3.7.0
            GIT_SHALLOW    TRUE)
        FetchContent_MakeAvailable(mpreal)
        set(MPREAL_INCLUDE_DIR ${mpreal_SOURCE_DIR})
    endif()

    # cpp_fuzz_mpfr reuses test/fuzz.cc with USE_MPFR defined — the single
    # source file carries an MPFR oracle in one compile mode and the
    # quadmath-only fast regression gate in the other.
    add_executable(cpp_fuzz_mpfr ./test/fuzz.cc)
    target_compile_options(cpp_fuzz_mpfr PRIVATE -O3 -fext-numeric-literals)
    target_compile_definitions(cpp_fuzz_mpfr PRIVATE USE_MPFR)
    target_include_directories(cpp_fuzz_mpfr PRIVATE ${MPREAL_INCLUDE_DIR})
    target_link_libraries(cpp_fuzz_mpfr PRIVATE multifloats quadmath MPFR::MPFR)
    add_test(NAME precision_mpfr_cpp COMMAND cpp_fuzz_mpfr 10000 42)
endif()
